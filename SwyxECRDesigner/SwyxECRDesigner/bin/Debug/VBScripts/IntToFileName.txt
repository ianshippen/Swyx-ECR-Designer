Function IntToFileName(p_int)
	Dim result

	result = ""

	Select Case p_int
		Case 1
			result = "One.wav"

		Case 2
			result = "Two.wav"

		Case 3
			result = "Three.wav"

		Case 4
			result = "Four.wav"

		Case 5
			result = "Five.wav"

		Case 6
			result = "Six.wav"

		Case 7
			result = "Seven.wav"

		Case 8
			result = "Eight.wav"

		Case 9
			result = "Nine.wav"

		Case else
			result = "Nine.wav"

	End Select

	IntToFileName = result
End Function
