Sub TimestampEx(byref p_table, byref p_myKey, byval p_callid, byref p_fieldName, byref p_ivrPath, byval p_cascadeLevel, byval p_attempt)
	DBExecute "EXEC SP_TIMESTAMP_EX " & WrapInSingleQuotes(p_table) & ", " & WrapInSingleQuotes(p_myKey) & ", " & p_callId & ", " & WrapInSingleQuotes(p_fieldName) & ", " & WrapInSingleQuotes(p_ivrPath) & ", " & p_cascadeLevel & ", " & p_attempt
End Sub
